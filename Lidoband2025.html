  fixed = LyingTime ~ four * TRT,
  random = ~ 1 | Cow,
  data = fourmodel
)

mod_ar1 <- lme(
  fixed = LyingTime ~ four * TRT,
  random = ~ 1 | Cow,
  correlation = corAR1(form = ~ TimeNum | Cow),
  data = fourmodel
)

mod_rs <- lme(
